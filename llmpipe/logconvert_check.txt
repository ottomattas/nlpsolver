EXAMPLE 5: Dogs have tails (normal rule, confidence 0.99)
----------------------------------------------------------------------
INPUT:
[and,
  [@id,
    S1,
    [and,
      [holds,
        W0,
        [forall,
          X,
          [implies, [isa,dog,X], [normally, [exists, Y, [and, [isa,tail,Y], [have,X,Y]]]]]
        ]
      ],
      [@p,S1,0.99]
    ]]]

OUTPUT:
[{@name: sent_S1,
  @logic: [[-isa,dog,?:X], [normally, [exists, Y, [and, [isa,tail,Y], [have,?:X,Y]]]]],
  @confidence: 0.99
  }]

EXAMPLE 6: Dogs do not have tails (strict negation rule)
----------------------------------------------------------------------
INPUT:
[and,

    S1,

[{@name: sent_S1, @logic: [[-isa,dog,?:X], [-isa,tail,?:Y], [-have,?:X,?:Y]]}]

======================================================================
EXAMPLE 7: Transitivity: X connected Y, Y connected Z => X connected Z
----------------------------------------------------------------------
INPUT:
[and,
  [@id,
    S1,
    [holds,
      W0,
      [forall,
        X,
        [forall,
          Y,
          [forall,
            Z,
            [implies,
              [and, [is_rel2,connected_to,X,Y], [is_rel2,connected_to,Y,Z]],
              [is_rel2,connected_to,X,Z]
            ]]]]]]]

OUTPUT:
[{@name: sent_S1,
  @logic: [[-is_rel2,connected_to,?:X,?:Y],
      [-is_rel2,connected_to,?:Y,?:Z],
      [is_rel2,connected_to,?:X,?:Z]
    ]}]
